function getPmids() { return fetch('https://cors-anywhere.herokuapp.com/https://eutils.ncbi.nlm.nih.gov/entrez/eutils/esearch.fcgi?db=pubmed&term=COVID-19') .then(response => response.json()) .then(data => data.esearchresult.idlist); } function getSummaries(pmids) { return fetch('https://cors-anywhere.herokuapp.com/https://eutils.ncbi.nlm.nih.gov/entrez/eutils/esummary.fcgi?db=pubmed&id=' + pmids.join(',') + '&retmode=xml') .then(response => response.text()); }
